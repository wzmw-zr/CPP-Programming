# 类型转换构造函数

在有参构造函数中的**只有一个参数的构造函数被称为类型转换构造函数。**

![](./constructor_destructor.png)

```c++
#include<iostream>
#include<string>
#include<vector>
#include<queue>
#include<stack>
#include<algorithm>
using namespace std;

class BigInt {
public:
    BigInt() {}
    // 转换构造函数
    BigInt(int x) {
        num.push_back(x);
        process_digit();
    }
    // 友元方法
    friend ostream &operator<<(ostream &out, const BigInt &a);

private:
    vector<int> num;
    void process_digit() {
        for (int i = 0; i < num.size(); i++) {
            if (num[i] < 10) continue;
            if (i + 1 == num.size()) num.push_back(0);
            num[i + 1] = num[i] / 10;
            num[i] %= 10;
        }
    }
};

// 重载对于大整数的运算符
ostream &operator<<(ostream &out, const BigInt &a) {
    for (int i = a.num.size() - 1; i >= 0; i--) out << a.num[i];
    return out;
}

void func(BigInt a) {
    cout << "func : " << a << endl;
}

int main() {
    BigInt a;
    // 普通的转换构造函数
    a = 1234;
    cout << a << endl;
    // 函数传值过程中的转换构造函数的使用
    func(5678);
    return 0;
}
```
